#### 1230 ~ 1300
case = int(input())   

def fac(x):   
    return x * fac(x - 1) if x > 1 else 1   

for _ in range(case):   
    r, n = map(int, input().split())  # nCr 조합 연산 = n! * (n-r)!*r   
    print fac(n) // (fac(n - r) * fac(r))   

operator.mul 써서 해봤는데 오히려 느렸다. 위 블로그에서는 더 빨랐다는데 반복수가 큰 경우에 그런 듯 하다.
